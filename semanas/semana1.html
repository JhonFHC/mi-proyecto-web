Semana 1 - ¿Qué es GIT?

GIT es un sistema de control de versiones que guarda el historial de cambios de un proyecto, permitiendo trabajar en equipo y retroceder a versiones anteriores.

Detalles y explicación

GIT guarda un historial de versiones llamado Snapshot o Commit, que permite almacenar solo los cambios realizados. Esto facilita trabajar sin conexión y sincronizar después con repositorios remotos.

Internamente, GIT maneja:

  • Working Directory: Archivos editables del proyecto.
  • Staging Area: Cambios seleccionados para guardar.
  • Repositorio: Lugar donde se almacenan las versiones.

¿Qué es una rama (Branch)?

Una rama permite separar el desarrollo para probar ideas sin afectar la rama principal, pudiendo luego unir los cambios.

JFrame Form


package Vista;

import javax.swing.JOptionPane;
import javax.swing.*;

public class DescuentoCamisa extends JFrame {

    private JButton btnCalcular, btnNuevo, btnSalir;
    private JTextField txtPrecio, txtCantidad;
    private JTextArea txtSalida;

    public DescuentoCamisa() {
        initComponents();
    }

    private void initComponents() {
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setTitle("Descuento de camisa");

        btnCalcular = new JButton("Calcular");
        btnNuevo = new JButton("Nuevo");
        btnSalir = new JButton("Salir");
        txtPrecio = new JTextField(10);
        txtCantidad = new JTextField(10);
        txtSalida = new JTextArea(8, 25);
        txtSalida.setEditable(false);

        JLabel lblTitulo = new JLabel("Descuento de camisa");
        JLabel lblPrecio = new JLabel("Precio de la camisa");
        JLabel lblCantidad = new JLabel("Cantidad");

        JPanel panel = new JPanel();
        panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));

        panel.add(lblTitulo);
        panel.add(Box.createVerticalStrut(10));

        JPanel filaPrecio = new JPanel();
        filaPrecio.add(lblPrecio);
        filaPrecio.add(txtPrecio);
        panel.add(filaPrecio);

        JPanel filaCantidad = new JPanel();
        filaCantidad.add(lblCantidad);
        filaCantidad.add(txtCantidad);
        panel.add(filaCantidad);

        JPanel filaBotones = new JPanel();
        filaBotones.add(btnCalcular);
        filaBotones.add(btnNuevo);
        filaBotones.add(btnSalir);
        panel.add(filaBotones);

        JScrollPane scrollSalida = new JScrollPane(txtSalida);
        panel.add(scrollSalida);

        add(panel);
        pack();
        setLocationRelativeTo(null);

        // Listeners
        btnCalcular.addActionListener(e -> calcularDescuento());
        btnNuevo.addActionListener(e -> limpiarCampos());
        btnSalir.addActionListener(e -> System.exit(0));
    }

    private void calcularDescuento() {
        try {
            float precio = Float.parseFloat(txtPrecio.getText().trim());
            int cantidad = Integer.parseInt(txtCantidad.getText().trim());
            if (precio <= 0 || cantidad <= 0) {
                JOptionPane.showMessageDialog(this, "Ingrese valores mayores a cero");
                return;
            }
            float importe = precio * cantidad;
            float descuento1 = importe * 0.07f;
            float descuento2 = (importe - descuento1) * 0.07f;
            float descuentoTotal = descuento1 + descuento2;
            float importePagar = importe - descuentoTotal;

            txtSalida.setText("Detalle de la compra:\n");
            txtSalida.append("----------------------------\n");
            txtSalida.append(String.format("Importe de la compra: %.2f\n", importe));
            txtSalida.append(String.format("Primer descuento (7%%): %.2f\n", descuento1));
            txtSalida.append(String.format("Segundo descuento (7%% sobre restante): %.2f\n", descuento2));
            txtSalida.append(String.format("Descuento total: %.2f\n", descuentoTotal));
            txtSalida.append(String.format("Importe a pagar: %.2f\n", importePagar));
        } catch (NumberFormatException ex) {
            JOptionPane.showMessageDialog(this, "Ingrese valores numéricos válidos");
        }
    }

    private void limpiarCampos() {
        txtPrecio.setText("");
        txtCantidad.setText("");
        txtSalida.setText("");
        txtPrecio.requestFocus();
    }
}
