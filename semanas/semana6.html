Formulario de Login

Interfaz principal de autenticación con validaciones en tiempo real, placeholders dinámicos y control de seguridad por intentos fallidos.

Clase FormLogin

public class FormLogin extends javax.swing.JFrame {
    private int intentosFallidos = 0;
    private static final int MAX_INTENTOS = 3;

    public FormLogin() {
        initComponents();
        setLocationRelativeTo(null);
        agregarValidacionTiempoReal();
        agregarPlaceholders();
    }

    // Validaciones principales
    private boolean validarCamposLogin() {
        String usuario = txtUsuario.getText().trim();
        String contrasenia = String.valueOf(txtContrasenia.getPassword());
        
        if (usuario.isEmpty() || usuario.equals("Ingrese su usuario")) {
            mostrarError("❌ El campo USUARIO es obligatorio");
            return false;
        }
        
        if (contrasenia.isEmpty() || contrasenia.equals("••••")) {
            mostrarError("❌ El campo CONTRASEÑA es obligatorio");
            return false;
        }
        
        if (contrasenia.length() < 4) {
            mostrarError("❌ La contraseña debe tener al menos 4 caracteres");
            return false;
        }
        
        if (usuario.contains(" ")) {
            mostrarError("❌ El usuario no puede contener espacios");
            return false;
        }
        
        return true;
    }

    // Control de intentos
    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {
        if (validarCamposLogin() && intentosFallidos < MAX_INTENTOS) {
            CLogin objetoLogin = new CLogin();
            boolean loginExitoso = objetoLogin.validaUsuario(txtUsuario, txtContrasenia);
            
            if (!loginExitoso) {
                intentosFallidos++;
                if (intentosFallidos >= MAX_INTENTOS) {
                    JOptionPane.showMessageDialog(this,
                        "❌ Demasiados intentos fallidos\nLa aplicación se cerrará.",
                        "Seguridad", JOptionPane.ERROR_MESSAGE);
                    System.exit(0);
                }
            }
        }
    }
}
// Navegación fluida entre ventanas
private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {
    FormSignUp SignupFrame = new FormSignUp();
    SignupFrame.setVisible(true);
    SignupFrame.pack();
    SignupFrame.setLocationRelativeTo(null);
    this.dispose(); // Liberar recursos
}

Sistema de Registro

Formulario complementario para el registro de nuevos usuarios con diferentes roles y validaciones específicas.

Clase FormSignUp

public class FormSignUp extends javax.swing.JFrame {

    public FormSignUp() {
        initComponents();
        setLocationRelativeTo(null);
        // Configurar roles disponibles
        comboRol.setModel(new javax.swing.DefaultComboBoxModel<>(
            new String[] { "estudiante", "profesor" }
        ));
    }

    private void btnRegistrarActionPerformed(java.awt.event.ActionEvent evt) {
        CRegistro objetoRegistro = new CRegistro();
        boolean registroExitoso = objetoRegistro.registrarUsuario(
            txtNombre, txtUsuario, txtContrasenia1, txtEmail, comboRol
        );
        
        if (registroExitoso) {
            // Volver al login después del registro
            FormLogin LoginFrame = new FormLogin();
            LoginFrame.setVisible(true);
            LoginFrame.setLocationRelativeTo(null);
            this.dispose();
        }
    }

    private void btnVolverActionPerformed(java.awt.event.ActionEvent evt) {
        FormLogin LoginFrame = new FormLogin();
        LoginFrame.setVisible(true);
        LoginFrame.setLocationRelativeTo(null);
        this.dispose();
    }
}
  • Campos Obligatorios: Todos los campos deben estar completos
  • Formato de Email: Validación de estructura de correo electrónico
  • Longitud Mínima: Contraseña de al menos 4 caracteres
  • Usuario Único: Verificación de usuario no existente
  • Sin Espacios: Usuario sin caracteres de espacio
  • Arquitectura MVC: Separación clara entre vista y lógica
  • Base de Datos: Integración con MySQL para persistencia
  • Encriptación: Contraseñas almacenadas de forma segura
  • Manejo de Errores: Excepciones controladas apropiadamente
  • Interfaz Responsive: Adaptable a diferentes resoluciones
