Semana 4 - Arquitectura MVC

Descubre el patrón de diseño Modelo-Vista-Controlador que separa la lógica de negocio, la interfaz de usuario y el control de flujo en aplicaciones Java.

Estructura del Patrón MVC

MODELO

Gestión de datos y lógica de negocio

VISTA

Interfaz de usuario y presentación

CONTROLADOR

Coordinación entre Modelo y Vista

Fundamentos del MVC

El patrón MVC separa las responsabilidades de una aplicación en tres componentes principales, facilitando el mantenimiento, testing y escalabilidad del código.

  • Modelo (Model): Representa los datos y la lógica de negocio
  • Vista (View): Maneja la presentación y la interfaz de usuario
  • Controlador (Controller): Gestiona las interacciones del usuario y coordina Modelo-Vista
  • Usuario → Vista: El usuario interactúa con la interfaz
  • Vista → Controlador: La vista notifica al controlador
  • Controlador → Modelo: El controlador actualiza el modelo
  • Modelo → Vista: El modelo notifica cambios a la vista
  • Vista → Usuario: La vista se actualiza y muestra resultados

Implementación Práctica

Ejemplos de implementación MVC en aplicaciones Java con login, conversión de unidades y cálculo de figuras geométricas.

Controlador de Figuras

public class FiguraControlador {
    private Figura modelo;
    private final CuadradoVista vista;

    public FiguraControlador(CuadradoVista vista) {
        this.vista = vista;
        
        // Conectar eventos de la vista
        this.vista.agregarCalcularListener(e -> calcular());
        this.vista.agregarNuevoListener(e -> vista.limpiarCampos());
    }

    public void calcular() {
        double lado = vista.obtenerLado();
        String figura = vista.getFiguraSeleccionada();

        // Crear modelo según figura seleccionada
        switch (figura) {
            case "cuadrado": modelo = new Cuadrado(lado); break;
            case "circulo": modelo = new Circulo(lado); break;
            case "triangulo": modelo = new Triangulo(lado); break;
        }

        // Usar modelo para cálculos
        double area = modelo.calcularArea();
        double perimetro = modelo.calcularPerimetro();
        
        // Actualizar vista con resultados
        vista.mostrarResultados(area, perimetro);
    }
}

Controlador de Conversión

public class ConversionControlador {
    private ConversionVista vista;
    private Usuario usuario;

    public ConversionControlador(ConversionVista vista, Usuario usuario) {
        this.vista = vista;
        this.usuario = usuario;

        // Configurar vista según rol
        vista.setRol(usuario.getRol());
        
        // Conectar eventos
        vista.agregarCalcularListener(e -> calcular());
        vista.agregarGuardarSalirListener(e -> guardarYSalir());
    }

    private void calcular() {
        double valor = Double.parseDouble(vista.getValor());
        String origen = vista.getUnidadOrigen();
        String destino = vista.getUnidadDestino();
        
        // Lógica de conversión en el controlador
        double resultado = convertir(valor, origen, destino);
        vista.setResultado(valor + " " + origen + " = " + resultado + " " + destino);
    }

    private double convertir(double valor, String origen, String destino) {
        // Implementación de conversión de unidades
        double valorEnMetros = switch (origen) {
            case "cm" -> valor / 100;
            case "mm" -> valor / 1000;
            case "km" -> valor * 1000;
            default -> valor;
        };

        return switch (destino) {
            case "cm" -> valorEnMetros * 100;
            case "mm" -> valorEnMetros * 1000;
            case "km" -> valorEnMetros / 1000;
            default -> valorEnMetros;
        };
    }
}

Controlador de Login

public class LoginControlador {
    private final LoginVista loginVista;
    private final List usuarios;

    public LoginControlador(LoginVista vista, List usuarios) {
        this.loginVista = vista;
        this.usuarios = usuarios;

        // Conectar evento de login
        this.loginVista.agregarLoginListener(e -> login());
    }

    private void login() {
        String usuario = loginVista.getUsuario();
        String password = loginVista.getPassword();

        // Validar credenciales en el modelo
        Usuario userEncontrado = usuarios.stream()
                .filter(u -> u.getUsername().equals(usuario) 
                          && u.getPassword().equals(password))
                .findFirst().orElse(null);

        if (userEncontrado == null) {
            loginVista.setMensaje("Credenciales incorrectas");
        } else {
            loginVista.setMensaje("Bienvenido " + userEncontrado.getUsername());
            abrirConversionVista(userEncontrado);
        }
    }
}
