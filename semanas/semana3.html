Descubre los principios fundamentales de la POO aplicados en Java: encapsulamiento, herencia, polimorfismo y abstracción a través de ejemplos prácticos.

Fundamentos de POO

La Programación Orientada a Objetos organiza el código alrededor de objetos que combinan datos y comportamientos, facilitando la reutilización y mantenimiento del software.

  • Encapsulamiento: Ocultar los datos internos y exponer solo lo necesario
  • Herencia: Crear nuevas clases basadas en clases existentes
  • Polimorfismo: Mismo método, diferentes comportamientos
  • Abstracción: Ocultar la complejidad y mostrar solo lo esencial
// Definición de una clase en Java
public class Producto {
    // Atributos (encapsulados)
    private String nombre;
    private double precio;
    private int stock;
    
    // Constructor
    public Producto(String nombre, double precio, int stock) {
        this.nombre = nombre;
        this.precio = precio;
        this.stock = stock;
    }
    
    // Métodos getter y setter
    public String getNombre() { return nombre; }
    public void setNombre(String nombre) { this.nombre = nombre; }
    
    public double getPrecio() { return precio; }
    public void setPrecio(double precio) { this.precio = precio; }
    
    // Método de negocio
    public double calcularTotal(int cantidad) {
        return precio * cantidad;
    }
}

Ejemplos Prácticos

Implementación de POO en aplicaciones Java con interfaces gráficas para gestión de ofertas y cálculos comerciales.

Clase COfertaProducto

public class COfertaProducto {
    // Atributos encapsulados
    private float precioDocena;
    private int cantidadDocenas;
    
    // Métodos de cálculo encapsulados
    public float calcularImporteCompra() {
        return precioDocena * cantidadDocenas;
    }
    
    public float calcularDescuento() {
        float importe = calcularImporteCompra();
        return (cantidadDocenas >= 10) ? 
            importe * 0.20f : importe * 0.10f;
    }
    
    public float calcularImportePagar() {
        return calcularImporteCompra() - calcularDescuento();
    }
    
    public int calcularLapiceros() {
        return (calcularImportePagar() >= 200) ? 
            cantidadDocenas * 2 : 0;
    }
    
    // Getters y Setters
    public void setPrecioDocena(float precio) { 
        this.precioDocena = precio; 
    }
    public void setCantidadDocenas(int cantidad) { 
        this.cantidadDocenas = cantidad; 
    }
}

Clase COfertaTransporte

public class COfertaTransporte {
    // Atributos con encapsulamiento
    private String turno;
    private int cantidadPasajes;
    private final float precioPasaje = 37.5f;
    
    // Métodos de negocio
    public float calcularImporteCompra() {
        return precioPasaje * cantidadPasajes;
    }
    
    public float calcularDescuento() {
        float importe = calcularImporteCompra();
        float porcentaje = (cantidadPasajes >= 15) ? 0.08f : 0.05f;
        return importe * porcentaje;
    }
    
    public float calcularImportePagar() {
        return calcularImporteCompra() - calcularDescuento();
    }
    
    public int calcularCaramelos() {
        return (calcularImportePagar() > 200) ? 
            2 * cantidadPasajes : 0;
    }
    
    // Getters y Setters
    public void setTurno(String turno) { this.turno = turno; }
    public void setCantidadPasajes(int cantidad) { 
        this.cantidadPasajes = cantidad; 
    }
}

Clase CFormulario

public class CFormulario extends JFrame {
    private COfertaProducto oferta = new COfertaProducto();
    
    private void btnCalcularActionPerformed() {
        // Obtener datos de la interfaz
        float precio = Float.parseFloat(txtPrecio.getText());
        int cantidad = Integer.parseInt(txtCantidad.getText());
        
        // Usar el objeto POO
        oferta.setPrecioDocena(precio);
        oferta.setCantidadDocenas(cantidad);
        
        // Calcular resultados
        float importeCompra = oferta.calcularImporteCompra();
        float descuento = oferta.calcularDescuento();
        float importePagar = oferta.calcularImportePagar();
        int lapiceros = oferta.calcularLapiceros();
        
        // Mostrar resultados
        mostrarResultados(importeCompra, descuento, 
                         importePagar, lapiceros);
    }
    
    private void validarCampos() {
        // Validación encapsulada en la interfaz
        boolean habilitar = !txtPrecio.getText().isEmpty() 
                         && !txtCantidad.getText().isEmpty();
        btnCalcular.setEnabled(habilitar);
    }
}
